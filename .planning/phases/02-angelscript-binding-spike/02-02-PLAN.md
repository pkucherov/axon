    ```python
    """Spike tests for the vendored tree-sitter-angelscript binding.

    These tests validate INFRA-06: the binding is importable and the
    grammar can parse all four D-07 construct categories from UnrealAngel
    AngelScript syntax.

    Per D-07 (CONTEXT.md):
    1. Function declarations (standalone and method)
    2. Class declarations with inheritance
    3. UE5-style annotations (UCLASS, UFUNCTION-equivalent)
    4. #include / import directives

    Note on preprocessor_directive: The dimitrijejankov grammar treats
    preprocessor directives as a catch-all terminal token (no structured
    child nodes). #include path extraction in Phase 4 must use
    node.text string parsing, not node.children traversal.
    """

    from __future__ import annotations

    import pytest


    def test_binding_importable() -> None:
        """INFRA-06: binding is importable and returns a valid Language object."""
        import tree_sitter_angelscript as tsas
        from tree_sitter import Language

        lang = Language(tsas.language())
        assert lang is not None


    def test_parse_class_with_inheritance() -> None:
        """D-07 criterion 2 and 3: class declaration with inheritance + UCLASS macro."""
        import tree_sitter_angelscript as tsas
        from tree_sitter import Language, Parser

        parser = Parser(Language(tsas.language()))
        source = b"UCLASS(BlueprintType)\nclass AMyActor : AActor\n{\n}"
        tree = parser.parse(source)

        assert tree.root_node.type == "program"
        classes = [n for n in tree.root_node.children if n.type == "class_declaration"]
        assert len(classes) == 1, (
            f"Expected 1 class_declaration, got {len(classes)}. "
            f"Root children: {[n.type for n in tree.root_node.children]}"
        )


    def test_parse_ufunction() -> None:
        """D-07 criterion 1 and 3: global function declaration + UFUNCTION macro."""
        import tree_sitter_angelscript as tsas
        from tree_sitter import Language, Parser

        parser = Parser(Language(tsas.language()))
        source = b"UFUNCTION(BlueprintCallable)\nvoid MyFunc()\n{\n}"
        tree = parser.parse(source)

        assert tree.root_node.type == "program"
        funcs = [n for n in tree.root_node.children if n.type == "global_function_declaration"]
        assert len(funcs) == 1, (
            f"Expected 1 global_function_declaration, got {len(funcs)}. "
            f"Root children: {[n.type for n in tree.root_node.children]}"
        )


    def test_parse_import() -> None:
        """D-07 criterion 4: import directive."""
        import tree_sitter_angelscript as tsas
        from tree_sitter import Language, Parser

        parser = Parser(Language(tsas.language()))
        source = b"import MyModule.SomeFile"
        tree = parser.parse(source)

        assert tree.root_node.type == "program"
        imports = [n for n in tree.root_node.children if n.type == "import_statement"]
        assert len(imports) == 1, (
            f"Expected 1 import_statement, got {len(imports)}. "
            f"Root children: {[n.type for n in tree.root_node.children]}"
        )


    def test_parse_preprocessor_include() -> None:
        """D-07 criterion 4: preprocessor #include directive.

        Note: preprocessor_directive is a catch-all terminal token in the
        dimitrijejankov grammar. The include path is embedded in the node
        text and must be extracted via string parsing in Phase 4 —
        not via child node traversal.
        """
        import tree_sitter_angelscript as tsas
        from tree_sitter import Language, Parser

        parser = Parser(Language(tsas.language()))
        source = b'#include "SomeFile.as"'
        tree = parser.parse(source)

        assert tree.root_node.type == "program"
        directives = [n for n in tree.root_node.children if n.type == "preprocessor_directive"]
        assert len(directives) == 1, (
            f"Expected 1 preprocessor_directive, got {len(directives)}. "
            f"Root children: {[n.type for n in tree.root_node.children]}"
        )
        assert b"SomeFile.as" in directives[0].text, (
            f"Expected 'SomeFile.as' in directive text, got: {directives[0].text}"
        )
    ```
